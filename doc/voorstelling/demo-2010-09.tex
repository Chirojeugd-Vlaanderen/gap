\overlays{7}
{
\begin{slide}{Wat verandert er in Kipadmin?}

\begin{itemize}
\fromSlide{2}{\item Automatische aanmaak van `aansluitingen' voor leden waarvan de probeerperiode voorbij is}
\fromSlide{3}{\item Wijzigingen van naam, adres, geboortedatum, telefoonnummer, \ldots van personen met AD-nummer, komen (quasi) direct door}
\fromSlide{4}{\item Dubbelpuntabonnementen en verzekering loonverlies komen (quasi) direct door}
\fromSlide{5} {\item \'Alle leden in Kipadmin; niet enkel keti's en aspi's}
\fromSlide{6} {\item Aansluitingen en verzekering loonverlies m\'oeten nu via GAP}
\fromSlide{7} {\item Bijwerken van de stratenlijst zal op het secretariaat moeten gebeuren}
\end{itemize}

\end{slide}
